Quick Start

Running in 30 seconds

install
# Clone & install git clone https://github.com/bizernam-cyber/whale-tracker-agent.git cd whale-tracker-agent pip install -e .
usage
# Watch a whale whale-track watch 0xd8dA6BF2...6045 --label "Vitalik" # Scan recent activity whale-track scan 0xd8dA...6045 --days 7 # Real-time monitor whale-track monitor --chain ethereum --min-amount 100
python api
import asyncio from whale_tracker.agent import WhaleTracker async def main(): tracker = WhaleTracker() await tracker.watch( "0xd8dA...6045", chain="ethereum", label="Vitalik" ) result = await tracker.scan( "0xd8dA...6045", days=7 ) print(result["patterns"]) asyncio.run(main())
